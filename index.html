import random def generate_random_amounts(count, min_val, max_val): """ 生成指定数量的随机金额，并计算平均值。 参数: count (int): 生成的随机金额数量 min_val (float): 金额最小值 max_val (float): 金额最大值 返回: dict: 包含随机金额列表和平均值的字典 """ # 1. 修正金额范围逻辑：确保 min_val <= max_val if min_val > max_val: # 如果用户输入调转，自动交换，避免报错或生成空集 min_val, max_val = max_val, min_val print(f"提示：检测到范围输入顺序有误，已自动调整为 [{min_val}, {max_val}]") if min_val == max_val: # 如果范围相同，直接返回固定值 amounts = [round(min_val, 2) for _ in range(count)] else: # 2. 生成随机金额：使用 uniform 生成浮点数，并保留两位小数 amounts = [round(random.uniform(min_val, max_val), 2) for _ in range(count)] # 3. 计算平均值 if not amounts: average_val = 0.0 else: total_sum = sum(amounts) average_val = round(total_sum / len(amounts), 2) return { "amounts": amounts, "average": average_val, "total": round(total_sum, 2) } # --- 测试示例 --- if __name__ == "__main__": # 模拟用户输入：生成10个金额，范围在 10.0 到 100.0 之间 # 注意：此处不使用任何具体购物网站链接，仅作为数值演示 num_envelopes = 10 min_money = 10.0 max_money = 100.0 # 故意测试范围调转的情况 (例如输入 max=10, min=100) # result = generate_random_amounts(num_envelopes, 100.0, 10.0) result = generate_random_amounts(num_envelopes, min_money, max_money) print(f"生成的随机金额列表: {result['amounts']}") print(f"总金额: {result['total']}") print(f"平均金额: {result['average']}")
